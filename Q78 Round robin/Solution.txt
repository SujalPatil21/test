class Solution {
    public int findKthPositive(int[] arr, int k) {

int count = 0;

        HashSet<Integer> set = new HashSet<>();
        for(int i = 0 ; i < arr.length ; i++){

            set.add(arr[i]);
        }
          int p = 1;

        while(true){    


            if(!set.contains(p)){

                count++;

            }
            if(count == k){

                return p;
            }
            p++;
        }
   
    }
}
